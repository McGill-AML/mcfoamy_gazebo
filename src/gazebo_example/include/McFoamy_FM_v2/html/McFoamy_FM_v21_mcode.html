256% CALCULATIONS
257u = Vel(1);
258v = Vel(2);
259w = Vel(3);
260p = B_rate(1);
261q = B_rate(2);
262r = B_rate(3);
263
264% negative b/c it is from c.g. to propeller plane
265xthr = -CG(1)*1e-3; 
266ythr = -CG(2)*1e-3;
267zthr = -CG(3)*1e-3;
268
269vthr_x = u + q*zthr - r*ythr;
270vthr_y = v + r*xthr - p*zthr;
271vthr_z = w + p*ythr - q*xthr;
272
273%-------------------------------------------------------------
274V = sqrt(vthr_x^2 + vthr_y^2 + vthr_z^2); % total velocity
275Vyz = sqrt(vthr_y^2 + vthr_z^2); % in-plane velocity
276psiT = atan2(Vyz,abs(vthr_x)); % azimuth angle of thruster (0 to +90 deg.)
277deltaT = atan2(vthr_z,vthr_y); % in-plane angle (0 to +/- 180)
278
279if wIn < 1716 % Too slow to run the motor
280    wOut = 0;
281    J = 0;
282    CFx = 0;
283    CFy = 0;
284    CFz = 0;
285    CMx = 0;
286    CMy = 0;
287    CMz = 0;
288else
289    wOut = wIn;
290    J = V/(wOut/60*D); % advance ratio based on total velocity
291    
292    if vthr_x >= 0 % Forward flight
293        if J <= 1
294            CFx = interp2(J_pre, psiT_pre, CFx_pre, J, psiT);
295            CFy = interp2(J_pre, psiT_pre, CFy_pre, J, psiT);
296            CFz = 0;
297            CMx = interp2(J_pre, psiT_pre, CMx_pre, J, psiT);
298            CMy = interp2(J_pre, psiT_pre, CMy_pre, J, psiT);
299            CMz = interp2(J_pre, psiT_pre, CMz_pre, J, psiT);
300        else
301            CFx = interp1(psiT_pre(:,11), CFx_pre(:,11), psiT);
302            CFy = interp1(psiT_pre(:,11), CFy_pre(:,11), psiT);
303            CFz = 0;
304            CMx = interp1(psiT_pre(:,11), CMx_pre(:,11), psiT);
305            CMy = interp1(psiT_pre(:,11), CMy_pre(:,11), psiT);
306            CMz = interp1(psiT_pre(:,11), CMz_pre(:,11), psiT);            
307        end
308
309    else % Rearward flight
310        % A/c to Bart's paper Fig.8 & 9, for rearward flight, Cfx and Cmx have
311        % approx. static value
312        CFx = 0.15633;
313        CFy = 0;
314        CFz = 0;
315        CMx = -0.00938;
316        CMy = interp2(J_pre, psiT_pre, CMy_pre, J, psiT);
317        CMz = interp2(J_pre, psiT_pre, CMz_pre, J, psiT);    
318    end
319end

321% Transformation into UAV XYZ frame

323CFY = -CFy*cos(deltaT) + CFz*sin(deltaT);
324CFZ = -CFy*sin(deltaT) - CFz*cos(deltaT);
325CMX = CMx;
326CMY = -CMy*cos(deltaT) + CMz*sin(deltaT);
327CMZ = -CMy*sin(deltaT) - CMz*cos(deltaT);

329% Aerodynamic Forces and Moments
330FX = rho*(wOut/60)^2*D^4*CFX;
331FY = rho*(wOut/60)^2*D^4*CFY;
332FZ = rho*(wOut/60)^2*D^4*CFZ;
333MX = rho*(wOut/60)^2*D^5*CMX;
334MY = rho*(wOut/60)^2*D^5*CMY;
335MZ = rho*(wOut/60)^2*D^5*CMZ;

337% Induced Velocity at Propeller Plane
338V0 = 1/2*(1.59*(wOut/60)*D*sqrt(0.15633)); % Induced velocity in hover
339
340if vthr_x >= -0.2*V0 && CFX > 0 % if reverse velocity is greater than 20% of hover velocity
341    Vi0_avg = 1/2*(1.59*(wOut/60)*D*sqrt(CFX));
342    MX = 0.2*MX;
343else
344    Vi0_avg = 0;    
345end

347% Swirling Flow
348MX = 0.4*MX;
349
350ThrForce = [FX,0.5.*FY,0.5.*FZ];
351
352ThrMom = [MX,0.5.*MY,0.5.*MZ];

362%% PROPELLER SLIPSTREAM MODEL
363Rh = PropGeom(1)/2*1e-3;
364Rp = PropGeom(2)/2*1e-3;
365
366DesX = abs(Geom(5,5:end))*1e-3;
367DesR = sqrt(Geom(6,5:end).^2 + Geom(7,5:end).^2)*1e-3;
368
369VProp_Axial = zeros(1,length(DesX));
370VProp_Swirl = zeros(1,length(DesX));
371
372% Equations used from propeller slipstream paper
373x0 = 1.528*Rp;   % Eq. 11
374R0 = 0.74*Rp;   % Eq. 12
375D0 = 2*R0;
376Rm0 = 0.67*(R0 - Rh);
377V0 = 2*Vi0_avg*(1.46/1.59); % Efflux velocity from Vi0_avg
378
379for i = 1:length(DesX)
380    if DesX(i) < x0
381        VProp_Axial(i) = Vi0_avg*(1 + (DesX(i)/Rp)/sqrt(1 + (DesX(i)/Rp)^2));  % Eq. 1        
382
383    elseif DesX(i) < 1.7*D0
384        Vmax = V0*(1.24 - 0.0765*(DesX(i)-x0)/D0);  % Eq. 13
385        Rm = Rm0*(1 - 0.1294*(DesX(i) - x0)/D0);    % Eq. 14
386        VProp_Axial(i) = Vmax*exp(-((DesR(i)-Rm)/(0.8839*Rm0 + 0.1326*(DesX(i)-x0-R0)))^2);   % Eq. 15
387    
388    elseif DesX(i) < 4.25*D0
389        Vmax = V0*(1.37 - 0.1529*(DesX(i)-x0)/D0);  % Eq. 16
390        Rm = Rm0*(1.3 - 0.3059*(DesX(i) - x0)/D0);  % Eq. 17
391        VProp_Axial(i) = Vmax*exp(-((DesR(i)-Rm)/(0.5176*Rm0 + 0.2295*(DesX(i)-x0-R0)))^2);   % Eq. 18
392    
393    else
394        Vmax = V0*(0.89 - 0.04*(DesX(i)-x0)/D0);    % Eq. 19
395        VProp_Axial(i) = Vmax*exp(-(DesR(i)/(0.2411*(DesX(i)-x0)))^2);    % Eq. 21
396        
397    end
398    
399    if VProp_Axial(i) < 0.01
400        VProp_Axial(i) = 0; 
401    end
402end

459%--------------------------------------------------------------------------
460%                   Wing, Tail, Rudder and Body velocities
461%--------------------------------------------------------------------------
462vws_xcomp = u + q*zw - r*yws + Vp_w;
463vws_ycomp = v + r*xw - p*zw;
464vws_zcomp = w + p*yws - q*xw - Vp_w_swirl;
465vws = sqrt(vws_xcomp.^2 + vws_ycomp.^2 + vws_zcomp.^2);
466vws_xz = sqrt(vws_xcomp.^2 + vws_zcomp.^2);
467
468vwp_xcomp = u + q*zw - r*ywp + Vp_w;
469vwp_ycomp = v + r*xw - p*zw;
470vwp_zcomp = w + p*ywp - q*xw + Vp_w_swirl;
471vwp = sqrt(vwp_xcomp.^2 + vwp_ycomp.^2 + vwp_zcomp.^2);
472vwp_xz = sqrt(vwp_xcomp.^2 + vwp_zcomp.^2); 
473
474vts_xcomp = u + q*zt - r*yts + Vp_t;
475vts_ycomp = v + r*xt - p*zt;
476vts_zcomp = w + p*yts - q*xt - Vp_t_swirl;
477vts = sqrt(vts_xcomp.^2 + vts_ycomp.^2 + vts_zcomp.^2);
478vts_xz = sqrt(vts_xcomp.^2 + vts_zcomp.^2);
479
480vtp_xcomp = u + q*zt - r*ytp + Vp_t;
481vtp_ycomp = v + r*xt - p*zt;
482vtp_zcomp = w + p*ytp - q*xt + Vp_t_swirl;
483vtp = sqrt(vtp_xcomp.^2 + vtp_ycomp.^2 + vtp_zcomp.^2);
484vtp_xz = sqrt(vtp_xcomp.^2 + vtp_zcomp.^2);
485
486vr_xcomp = u + q*zr - r*yr + Vp_r;
487vr_ycomp = v + r*xr - p*zr + sign(zr + zCG).*Vp_r_swirl;
488vr_zcomp = w + p*yr - q*xr;
489vr = sqrt(vr_xcomp.^2 + vr_ycomp.^2 + vr_zcomp.^2);
490vr_xy = sqrt(vr_xcomp.^2 + vr_ycomp.^2);
491
492vB_xcomp = u + q*zB - r*yB + Vp_B;
493vB_ycomp = v + r*xB - p*zB + sign(zB + zCG).*Vp_B_swirl;
494vB_zcomp = w + p*yB - q*xB;
495vB = sqrt(vB_xcomp.^2 + vB_ycomp.^2 + vB_zcomp.^2);
496vB_xy = sqrt(vB_xcomp.^2 + vB_ycomp.^2);

502% range is -180 -> 180
503a_ws = atan2(vws_zcomp,vws_xcomp);
504a_wp = atan2(vwp_zcomp,vwp_xcomp);
505
506a_ts = atan2(vts_zcomp,vts_xcomp);
507a_tp = atan2(vtp_zcomp,vtp_xcomp);
508
509% Vertical AoA for rudder and body = atan(V_ycomp / V_xcomp)
510% range is -180 -> 180
511a_r = atan2(vr_ycomp,vr_xcomp);
512a_B  = atan2(vB_ycomp,vB_xcomp);

514%--------------------------------------------------------------------------
515%                   Wing lift and drag coefficient
516%-------------------------------------------------------------------------- 
517CN_ws = zeros(1,Nw);
518CL_ws = zeros(1,Nw);
519CD_ws = zeros(1,Nw);
520CM_ws = zeros(1,Nw);
521
522CN_wp = zeros(1,Nw);
523CL_wp = zeros(1,Nw);
524CD_wp = zeros(1,Nw);
525CM_wp = zeros(1,Nw);
526
527for i = 1:Nw
528    [CN_ws(i),CL_ws(i),CD_ws(i),CM_ws(i)] = McFoamy_Airfoil_Simplified(i,a_ws(i),Cwf(i),Cw(i),RAilDef,AR_w,Cd0,Cd90,alp0);
529    [CN_wp(i),CL_wp(i),CD_wp(i),CM_wp(i)] = McFoamy_Airfoil_Simplified(i,a_wp(i),Cwf(i),Cw(i),-RAilDef,AR_w,Cd0,Cd90,alp0);    
530end
531
532%--------------------------------------------------------------------------
533%                  Tail's lift and drag coefficient
534%--------------------------------------------------------------------------
535CN_ts = zeros(1,Nt);
536CL_ts = zeros(1,Nt);
537CD_ts = zeros(1,Nt);
538CM_ts = zeros(1,Nt);
539
540CN_tp = zeros(1,Nt);
541CL_tp = zeros(1,Nt);
542CD_tp = zeros(1,Nt);
543CM_tp = zeros(1,Nt);
544
545for i = 1:Nt    
546    [CN_ts(i),CL_ts(i),CD_ts(i),CM_ts(i)] = McFoamy_Airfoil_Simplified(i,a_ts(i),Ctf(i),Ct(i),ElevDef,AR_t,Cd0,Cd90,alp0);

574%--------------------------------------------------------------------------
575%                           Structural rods
576%--------------------------------------------------------------------------
577
578tol      = 1e-6;
579B_rate_p = B_rate';
580Vel_p    = Vel';
581CD_lg    = 1.1;
582D_rods   = zeros(3,1);
583M_rods   = zeros(3,1);
584
585for i = 1:size(rods,1)
586    
587    ai = rods(i,1:3)';  % pt1
588    bi = rods(i,4:6)';  % pt2
589    di = rods(i,7);     % diameter of the rod
590    ri = (ai + bi)/2;   % center of the rod w/r to the CG
591    li = bi - ai;       % length vector of the rod
592    Vi = Vel_p + cross(B_rate_p,ri)+[Vp_lg(i),0,0]';
593    sin_th = norm(cross(Vi,li))/max(norm(Vi),tol)/norm(li);
594    V_l_l = cross(cross(Vi,li),li);
595    
596    if Vi'*V_l_l < 0  % Vi pointing x forward, Nui pointing backward
597        Nui = V_l_l/max(norm(Vi)*norm(li)^2,tol);
598    else
599        Nui = -V_l_l/max(norm(Vi)*norm(li)^2,tol);
600    end
601    
602    Di = 0.5*rho*norm(Vi)^2*norm(li)*di*CD_lg*sin_th^2*Nui;
603    Mi = cross(ri,Di);
604    D_rods = D_rods+Di;
605    M_rods = M_rods+Mi;
606    
607end

609%--------------------------------------------------------------------------
610%                   Aerodynamic Forces & Moments
611%--------------------------------------------------------------------------
612% x-direction forces
613Fx_ws = 0.5*rho*bw.*Cw.*vws_xz.^2.*(CL_ws.*sin(a_ws) - CD_ws.*cos(a_ws));
614Fx_wp = 0.5*rho*bw.*Cw.*vwp_xz.^2.*(CL_wp.*sin(a_wp) - CD_wp.*cos(a_wp));
615Fx_ts = 0.5*rho*bt.*Ct.*vts_xz.^2.*(CL_ts.*sin(a_ts) - CD_ts.*cos(a_ts));
616Fx_tp = 0.5*rho*bt.*Ct.*vtp_xz.^2.*(CL_tp.*sin(a_tp) - CD_tp.*cos(a_tp));
617
618Fx_r = 0.5*rho*br.*Cr.*vr_xy.^2.*(CL_r.*sin(a_r) - CD_r.*cos(a_r));
619Fx_B = 0.5*rho*bB.*CB.*vB_xy.^2.*(CL_B.*sin(a_B) - CD_B.*cos(a_B));
620
621% y-direction forces
622Fy_ws = [0,0,0,0,0,0,0]; % INCLUDE FRICTION DRAG
623Fy_wp = [0,0,0,0,0,0,0]; % INCLUDE FRICTION DRAG
624Fy_ts = [0,0,0]; % INCLUDE FRICTION DRAG
625Fy_tp = [0,0,0]; % INCLUDE FRICTION DRAG
626
627Fy_r = 0.5*rho*br.*Cr.*vr_xy.^2.*(-CL_r.*cos(a_r) - CD_r.*sin(a_r));
628Fy_B = 0.5*rho*bB.*CB.*vB_xy.^2.*(-CL_B.*cos(a_B) - CD_B.*sin(a_B));
629
630% z-direction forces
631Fz_ws = 0.5*rho*bw.*Cw.*vws_xz.^2.*(-CL_ws.*cos(a_ws) - CD_ws.*sin(a_ws));
632Fz_wp = 0.5*rho*bw.*Cw.*vwp_xz.^2.*(-CL_wp.*cos(a_wp) - CD_wp.*sin(a_wp));
633Fz_ts = 0.5*rho*bt.*Ct.*vts_xz.^2.*(-CL_ts.*cos(a_ts) - CD_ts.*sin(a_ts));
634Fz_tp = 0.5*rho*bt.*Ct.*vtp_xz.^2.*(-CL_tp.*cos(a_tp) - CD_tp.*sin(a_tp));
635Fz_r = [0,0,0,0]; % INCLUDE FRICTION DRAG
636Fz_B = [0,0,0,0]; % INCLUDE FRICTION DRAG
637
638% x-direction moments
639Mx_ws = yws.*Fz_ws - zw.*Fy_ws;
640Mx_wp = ywp.*Fz_wp - zw.*Fy_wp;
641Mx_ts = yts.*Fz_ts - zt.*Fy_ts;
642Mx_tp = ytp.*Fz_tp - zt.*Fy_tp;
643Mx_r = yr.*Fz_r - zr.*Fy_r;
644Mx_B = yB.*Fz_B - zB.*Fy_B;
645Mx_Thr = yp*ThrForce(3) - zp*ThrForce(2);
646
647% y-direction moments
648My_ws = zw.*Fx_ws - xw.*Fz_ws + 0.5*rho*bw.*Cw.*vws_xz.^2.*Cw.*CM_ws;
649My_wp = zw.*Fx_wp - xw.*Fz_wp + 0.5*rho*bw.*Cw.*vwp_xz.^2.*Cw.*CM_wp;
650My_ts = zt.*Fx_ts - xt.*Fz_ts + 0.5*rho*bt.*Ct.*vts_xz.^2.*Ct.*CM_ts;
651My_tp = zt.*Fx_tp - xt.*Fz_tp + 0.5*rho*bt.*Ct.*vtp_xz.^2.*Ct.*CM_tp;
652My_r = zr.*Fx_r - xr.*Fz_r;
653My_B = zB.*Fx_B - xB.*Fz_B;
654My_Thr = zp*ThrForce(1) - xp*ThrForce(3);
655
656% z-direction moments
657Mz_ws = xw.*Fy_ws - yws.*Fx_ws;
658Mz_wp = xw.*Fy_wp - ywp.*Fx_wp;
659Mz_ts = xt.*Fy_ts - yts.*Fx_ts;
660Mz_tp = xt.*Fy_tp - ytp.*Fx_tp;
661Mz_r = xr.*Fy_r - yr.*Fx_r - 0.5*rho*br.*Cr.*vr_xy.^2.*Cr.*CM_r;  % WHY NEGATIVE SIGN???
662Mz_B = xB.*Fy_B - yB.*Fx_B - 0.5*rho*bB.*CB.*vB_xy.^2.*CB.*CM_B;  % WHY NEGATIVE SIGN???
663Mz_Thr = xp*ThrForce(2) - yp*ThrForce(1);
664
665% Total forces and moments in x, y and z directions
666Fx = sum(Fx_ws) + sum(Fx_wp) + sum(Fx_ts) + sum(Fx_tp) + sum(Fx_r) + sum(Fx_B) + ThrForce(1) + D_rods(1);
667Fy = sum(Fy_ws) + sum(Fy_wp) + sum(Fy_ts) + sum(Fy_tp) + sum(Fy_r) + sum(Fy_B) + ThrForce(2) + D_rods(2);
668Fz = sum(Fz_ws) + sum(Fz_wp) + sum(Fz_ts) + sum(Fz_tp) + sum(Fz_r) + sum(Fz_B) + ThrForce(3) + D_rods(3);
669
670Mx = sum(Mx_ws) + sum(Mx_wp) + sum(Mx_ts) + sum(Mx_tp) + sum(Mx_r) + sum(Mx_B) + Mx_Thr + ThrMom(1) + GyroMom(1)+ M_rods(1);
671My = sum(My_ws) + sum(My_wp) + sum(My_ts) + sum(My_tp) + sum(My_r) + sum(My_B) + My_Thr + ThrMom(2) + GyroMom(2)+ M_rods(2);
672Mz = sum(Mz_ws) + sum(Mz_wp) + sum(Mz_ts) + sum(Mz_tp) + sum(Mz_r) + sum(Mz_B) + Mz_Thr + ThrMom(3) + GyroMom(3)+ M_rods(3);
