1function [ F ] = force_controller(r_a, r_ref_a, v_b, v_ref_r, C_ba, C_ra,dt,maneuver_switch,parameters,gains)
 2    persistent delta_h_i_a 
 3    if isempty(delta_h_i_a)
 4        delta_h_i_a = 0;
 5    end
 6    delta_h_a = -(r_ref_a(3)-r_a(3)); %Height error (m)
 7    delta_h_i_a = delta_h_i_a + delta_h_a * dt; %Height error integral (ms)
 8    if (maneuver_switch == true)
 9        delta_h_i_a = 0.0;
10    end    
11    delta_h_b = C_ba*[0;0;-delta_h_a];
12    delta_h_i_b = C_ba*[0;0;-delta_h_i_a];
13    a_des_b = gains.Kv * (C_ba * C_ra' * v_ref_r - v_b) + gains.Khp * delta_h_b + gains.Khi * delta_h_i_b;
14    V = norm(v_b);
15    F_aero = gains.K_aero*(parameters.Faero2*V^2 + parameters.Faero1*V + parameters.Faero0);
16    g_b = C_ba * [0;0;9.81];
17    F = (-parameters.m * g_b - F_aero + parameters.m * a_des_b)'* parameters.Fhat;
18end
