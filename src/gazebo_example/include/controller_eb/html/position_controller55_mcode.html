1function [C_da] = position_controller(r_a,r_ref_a,v_b,v_ref_r,C_ra,C_ba,parameters,gains)
 2    v_r = zeros(3,1);
 3    v_r = C_ra * C_ba' * v_b;
 4    theta_r = cross(parameters.Fhat,gains.Kpp * C_ra * (r_ref_a - r_a) + gains.Kpd * (v_ref_r - v_r));
 5    for lv1 = 1:3
 6        theta_r(lv1) = saturate(theta_r(lv1),-3.1415/4,3.1415/4);
 7    end
 8    rot1 = [1,0,0;0,cos(theta_r(1)),sin(theta_r(1));0,-sin(theta_r(1)),cos(theta_r(1))];
 9    rot2 = [cos(theta_r(2)),0,-sin(theta_r(2));0,1,0;sin(theta_r(2)),0,cos(theta_r(2))];
10    rot3 = [cos(theta_r(3)),sin(theta_r(3)),0;-sin(theta_r(3)),cos(theta_r(3)),0;0,0,1];
11    C_da = rot1 * rot2 * rot3 * C_ra;    
12end
