1function [M_b] = attitude_controller(C_ba, C_ra, C_da, omega_ra_r, omega_ba_b, parameters,gains)
 2    %Computes Desired Moments based on desired aircraft attitude
 3    C_db = C_da * C_ba';
 4    phi = acos((trace(C_db)-1)/2);
 5    if phi == 0 
 6        a = zeros(3,1);
 7    else
 8        a = [C_db(2,3)-C_db(3,2); C_db(3,1)-C_db(1,3); C_db(1,2)-C_db(2,1)]/(2*sin(phi));
 9    end
10    E_b = phi * a;
11    M_b = diag([parameters.Ix, parameters.Iy, parameters.Iz])*(gains.Kap * E_b + gains.Kad * (C_ba * C_ra' * omega_ra_r - omega_ba_b));
12end
